class Advanced_UI_Widget extends \Widget_Base {
public function get_name() { 'bespoke_ui'; }
public function get_title() { 'Bespoke Data Engine'; }

// Custom control registration logic
protected function register_controls() {
$this->start_controls_section('logic_settings', [
'label' => 'Engine Parameters'
]);
$this->add_control('source', [
'type' => \Controls_Manager::SELECT,
'options' => ['api' => 'REST']
]);
$this->end_controls_section();
}

// Specialized rendering
protected function render() {
$settings = $this->get_settings_for_display();
$data = apply_filters('custom_fetch', $settings);
include locate_template('views/bespoke.php');
}
}
